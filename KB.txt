def allowed_file(filename):
    return '.' in filename and filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS

def extract_text_from_pdf(file_path):
    """从PDF文件提取文本"""
    text = ""
    try:
        with open(file_path, 'rb') as file:
            pdf_reader = PyPDF2.PdfReader(file)
            for page in pdf_reader.pages:
                text += page.extract_text() + "\n"
    except Exception as e:
        print(f"❌ PDF解析失败: {e}")
    return text

def extract_text_from_docx(file_path):
    """从DOCX文件提取文本"""
    text = ""
    try:
        doc = Document(file_path)
        for paragraph in doc.paragraphs:
            text += paragraph.text + "\n"
        # 处理表格
        for table in doc.tables:
            for row in table.rows:
                for cell in row.cells:
                    text += cell.text + "\t"
                text += "\n"
    except Exception as e:
        print(f"❌ DOCX解析失败: {e}")
    return text

def detect_encoding(file_path):
    """检测文件编码"""
    if HAS_CHARDET:
        try:
            with open(file_path, 'rb') as file:
                raw_data = file.read()
                result = chardet.detect(raw_data)
                return result['encoding']
        except Exception:
            pass
    # 如果检测失败，返回默认编码
    return 'utf-8'

def extract_text_from_txt(file_path):
    """从TXT文件提取文本（自动检测编码）"""
    encodings = ['utf-8', 'gbk', 'gb2312', 'latin-1']
    
    # 如果有编码检测库，优先使用它
    if HAS_CHARDET:
        detected_encoding = detect_encoding(file_path)
        if detected_encoding:
            encodings.insert(0, detected_encoding)
    
    for encoding in encodings:
        try:
            with open(file_path, 'r', encoding=encoding) as file:
                return file.read()
        except UnicodeDecodeError:
            continue
        except Exception as e:
            print(f"❌ 使用编码 {encoding} 读取文件失败: {e}")
            continue
    
    print(f"❌ 无法识别文件编码: {file_path}")
    return ""

def extract_text_from_json(file_path):
    """从JSON文件提取文本"""
    try:
        with open(file_path, 'r', encoding='utf-8') as file:
            data = json.load(file)
            return json.dumps(data, ensure_ascii=False, indent=2)
    except Exception as e:
        print(f"❌ JSON解析失败: {e}")
    return ""

def extract_text_from_file(file_path):
    """根据文件扩展名提取文本"""
    ext = file_path.rsplit('.', 1)[1].lower()
    if ext == 'pdf':
        return extract_text_from_pdf(file_path)
    elif ext == 'docx':
        return extract_text_from_docx(file_path)
    elif ext in ['txt', 'py', 'js', 'html', 'css', 'csv', 'md']:
        return extract_text_from_txt(file_path)
    elif ext == 'json':
        return extract_text_from_json(file_path)
    else:
        return ""

def chunk_text(text, chunk_size=500, overlap=50):
    """将文本分块（完整修复版）"""
    chunks = []
    text = text.strip()
    if not text:
        return chunks
    
    # 按固定长度分块，带重叠
    start = 0
    text_length = len(text)
    while start < text_length:
        end = min(start + chunk_size, text_length)
        chunk = text[start:end]
        chunks.append(chunk)
        # 移动指针，考虑重叠
        start = end - overlap
        # 避免死循环
        if start >= text_length - overlap:
            break
    return chunks

def embed_text(text: str) -> np.ndarray:
    """获取归一化向量"""
    try:
        emb_result = llm_emb.create_embedding(text)
        vec = np.array(emb_result['data'][0]['embedding'], dtype=np.float32)
        norm = np.linalg.norm(vec)
        if norm > 0:
            vec /= norm
        return vec
    except Exception as e:
        print(f"❌ 文本嵌入失败: {e}")
        # 返回零向量作为fallback
        return np.zeros(EMBEDDING_DIMENSION, dtype=np.float32)

def add_knowledge(text: str, source: str = "manual"):
    """添加知识到知识库"""
    if not text.strip():
        return
    # 检查是否已存在相似内容（避免重复）
    text_normalized = re.sub(r'\s+', '', text.lower())
    for existing_text in documents:
        existing_normalized = re.sub(r'\s+', '', existing_text.lower())
        if text_normalized == existing_normalized:
            print(f"ℹ️ 知识已存在，跳过: {text[:50]}...")
            return
    
    try:
        vec = embed_text(text)
        vectorstore.add(np.array([vec], dtype=np.float32))
        # 添加来源信息
        full_text = f"[来源: {source}]\n{text}" if source != "manual" else text
        documents.append(full_text)
        print(f"✅ 已添加：{text[:50]}...")
    except Exception as e:
        print(f"❌ 添加知识失败: {e}")

def add_file_content(file_path: str):
    """添加文件内容到知识库"""
    print(f"📄 正在处理文件: {file_path}")
    try:
        text = extract_text_from_file(file_path)
        if not text.strip():
            print(f"⚠️ 文件为空或无法解析: {file_path}")
            return
        
        # 分块处理长文档
        chunks = chunk_text(text, chunk_size=500, overlap=50)
        print(f"   -> 将文件分成 {len(chunks)} 个块")
        
        added_count = 0
        for i, chunk in enumerate(chunks):
            if chunk.strip():
                # 添加文件来源标识
                chunk_with_source = f"[来源: {os.path.basename(file_path)}, 第{i+1}/{len(chunks)}块]\n{chunk}"
                add_knowledge(chunk_with_source, source=os.path.basename(file_path))
                added_count += 1
        
        print(f"✅ 文件 {file_path} 成功添加，共 {added_count} 个知识块")
    except Exception as e:
        print(f"❌ 处理文件失败: {e}")

def add_folder_content(folder_path: str):
    """添加文件夹中所有支持的文件"""
    print(f"📁 正在处理文件夹: {folder_path}")
    added_files = 0
    for root, dirs, files in os.walk(folder_path):
        for file in files:
            if allowed_file(file):
                file_path = os.path.join(root, file)
                add_file_content(file_path)
                added_files += 1
    print(f"✅ 文件夹处理完成，共处理 {added_files} 个文件")

def search(query: str, top_k=3, threshold=0.3) -> List[dict]:
    """
    高智商搜索功能
    返回包含相似度分数的结果
    """
    if vectorstore.ntotal == 0:
        return []
    
    try:
        q_vec = embed_text(query)
        scores, idx = vectorstore.search(np.array([q_vec], dtype=np.float32), min(top_k * 2, vectorstore.ntotal))
        
        results = []
        for score, i in zip(scores[0], idx[0]):
            if 0 <= i < len(documents) and score >= threshold:
                results.append({
                    "content": documents[i],
                    "similarity": float(score),
                    "index": int(i)
                })
        
        # 按相似度排序并限制数量
        results.sort(key=lambda x: x['similarity'], reverse=True)
        return results[:top_k]
    except Exception as e:
        print(f"🔍 检索失败: {e}")
        return []

def smart_search(query: str, top_k=3) -> List[str]:
    """
    智能搜索 - 结合关键词匹配和语义搜索
    """
    if vectorstore.ntotal == 0:
        return []
    
    # 1. 语义搜索
    semantic_results = search(query, top_k=top_k, threshold=0.0)
    
    # 2. 关键词搜索（作为补充）
    keyword_matches = []
    query_words = set(re.findall(r'\w+', query.lower()))
    
    for i, doc in enumerate(documents):
        doc_words = set(re.findall(r'\w+', doc.lower()))
        common_words = query_words.intersection(doc_words)
        if common_words:
            # 计算关键词匹配度（简单交集比例）
            match_ratio = len(common_words) / max(len(query_words), 1)
            keyword_matches.append((doc, match_ratio, i))
    
    # 按匹配度排序
    keyword_matches.sort(key=lambda x: x[1], reverse=True)
    keyword_results = [{"content": item[0], "similarity": item[1], "index": item[2]} for item in keyword_matches[:top_k]]
    
    # 合并结果，去重
    all_results = {}
    for result in semantic_results + keyword_results:
        content = result['content']
        if content not in all_results or all_results[content]['similarity'] < result['similarity']:
            all_results[content] = result
    
    # 按相似度排序并返回
    final_results = sorted(all_results.values(), key=lambda x: x['similarity'], reverse=True)[:top_k]
    return [result['content'] for result in final_results]

def answer(query: str, context: List[str]):
    """
    高智商回答函数，支持代码/命令与自然语言的区分
    """
    # 检查是否为代码/命令格式
    is_code_or_command = (
        query.startswith(('!', '.', '/', 'python ', 'pip ', 'ls ', 'cat ', 'grep ', 'echo ', 'mkdir ', 'rm ', 'cp ', 'mv '))
        or ('.' in query and not query.startswith(' ') and not query.endswith('.'))  # 如 'rag_chat.py'
        or query.count('.') > 1  # 多个小数点/路径分隔符
        or (' ' not in query.split('.')[0] and '.' in query)  # 'file.py' 类似格式
    )

    ctx_text = "\n".join(context) if context else "无相关知识"

    # Qwen 模型专用模板
    if is_code_or_command:
        prompt = f"""<|im_start|>system
你是一个专业的代码与命令解释助手，根据资料准确回答。<|im_end|>
<|im_start|>user
资料：{ctx_text}
命令/文件：{query}
请解释这个命令或文件。<|im_end|>
<|im_start|>assistant
"""
    else:
        prompt = f"""<|im_start|>system
你是一个专业的编程助手，根据提供的资料严谨回答问题，不知道就说不知道。<|im_end|>
<|im_start|>user
资料：{ctx_text}
问题：{query}<|im_end|>
<|im_start|>assistant
"""

    try:
        output = llm_chat.create_completion(
            prompt=prompt,
            max_tokens=800,
            temperature=0.1,
            stop=["<|im_end|>"],
            echo=False
        )
        return output["choices"][0]["text"].strip()
    except Exception as e:
        return f"❌ 生成失败：{str(e)}"

    if not os.path.exists(UPLOAD_FOLDER):

    print("   • /add_folder <文件夹路径>  - 添加文件夹中所有文件")
    print("   • /search <关键词>  - 搜索知识库")
    print("   • /save       - 手动保存知识库")

                if os.path.exists(file_path):
                    add_file_content(file_path)
                else:

            elif ipt.startswith("/add_folder "):
                folder_path = ipt[12:]
                if os.path.exists(folder_path) and os.path.isdir(folder_path):
                    add_folder_content(folder_path)
                else:
                    print(f"❌ 文件夹不存在: {folder_path}")
            elif ipt.startswith("/search "):
                query = ipt[8:]
                results = smart_search(query)
                if results:
                    print(f"🔍 搜索结果 (共{len(results)}条):")
                    for i, result in enumerate(results, 1):
                        # 显示来源和内容预览
                        lines = result.split('\n')
                        source_line = lines[0] if lines and lines[0].startswith('[来源:') else "[来源: 未知]"
                        content_preview = '\n'.join(lines[1:]) if len(lines) > 1 else result
                        print(f"  {i}. {source_line}")
                        print(f"     {content_preview[:100]}{'...' if len(content_preview) > 100 else ''}")
                else:
                    print("🔍 未找到相关内容")
            elif ipt == "/save":
                save_knowledge_base()
            elif ipt == "/status":
                print(f"📊 状态: {len(documents)} 条知识, {vectorstore.ntotal} 个向量, 维度 {EMBEDDING_DIMENSION}")

                ctx = smart_search(ipt)

            print("\n⚠️ 检测到中断信号")
            save_knowledge_base()
            print("👋 已保存知识库并退出")
            break
        except Exception as e:
            print(f"❌ 发生错误: {e}")
            print("💡 程序将继续运行，请继续输入")

    for agent_file in os.listdir(agents_dir):
        if agent_file.endswith('.yaml'):
            file_path = os.path.join(agents_dir, agent_file)
            try:
                with open(file_path, 'r', encoding='utf-8') as f:
                    config_data = yaml.safe_load(f)
                    agent_name = config_data.get('name', agent_file.split('.')[0])
                    agents[agent_name] = config_data
                    logger.info(f"Loaded agent: {agent_name} from {agent_file}")
            except Exception as e:
                logger.error(f"Error loading agent {agent_file}: {str(e)}")
    
    if not agents:
        logger.warning("No agents found. Creating default agent...")
        default_agent = {
            'name': 'DEFAULT_AGENT',
            'description': 'Default fallback agent',
            'prompt_template': 'You are a helpful AI assistant. User question: {query}'
        }
        agents['DEFAULT_AGENT'] = default_agent
    
    return agents

def chat_with_model(prompt):
    data = {
        "prompt": prompt,
        "temperature": 0.7,

import llama_cpp
import chromadb

# ======================== 1. 启动 llama.cpp（你的大模型）
llm = llama_cpp.Llama(
    model_path="qwe.gguf",
    n_ctx=2048,
    n_gpu_layers=29
)

# ======================== 2. 启动 llama.cpp 嵌入模型（你的 Qwen3-Embedding）
embed = llama_cpp.Llama(
    model_path="Qwen3-Embedding-0.6B-Q8_0.gguf",
    n_ctx=512,
    embedding=True
)

# ======================== 3. 连接你已有的向量数据库（不重建！）
client = chromadb.PersistentClient(path="chroma_db")
collection = client.get_collection("my_docs")

# ======================== 4. 真正合并：检索 + 回答 一体化
def ask(prompt):
    # 1. 生成问题向量（用 llama.cpp）
    query_embedding = embed.create_embedding(prompt)["data"][0]["embedding"]

    # 2. 从数据库查最相关内容
    result = collection.query(
        query_embeddings=[query_embedding],
        n_results=2
    )
    context = "\n".join(result["documents"][0])

    # 3. 给 llama.cpp 生成最终回答
    final_prompt = f"根据资料：{context}\n问题：{prompt}\n回答："
    return llm.create_completion(final_prompt)["choices"][0]["text"]

# ======================== 运行
while True:
    q = input("\n你问：")
    if q == "exit": break
    print("\nAI：", ask(q))

11.run_embedding.py
import os
import json
from llama_cpp import Llama

# ==========================================
# 1. 配置路径
# ==========================================
# 嵌入模型路径
EMBED_MODEL_PATH = "./Qwen3-Embedding-0.6B-Q8_0.gguf"

# 知识库文件夹
KNOWLEDGE_BASE_DIR = "./suiwen_kb"

# 输出向量数据库文件
OUTPUT_VECTOR_DB = "./vector_db.json"

# ==========================================
# 2. 加载嵌入模型
# ==========================================
print(f"正在加载嵌入模型: {EMBED_MODEL_PATH} ...")
llm = Llama(
    model_path=EMBED_MODEL_PATH,
    embedding=True,      # ⚠️ 必须开启：告诉模型我们要提取向量
    n_ctx=8192,          # 上下文窗口大小
    verbose=False        # 关闭详细日志
)

# ==========================================
# 3. 处理知识库
# ==========================================
print(f"正在处理知识库: {KNOWLEDGE_BASE_DIR} ...")

# 用于存储向量和文本的列表
vector_db = []

# 遍历知识库文件夹中的所有文件
for filename in os.listdir(KNOWLEDGE_BASE_DIR):
    # 只处理 .txt 和 .md 文件
    if filename.endswith(".txt") or filename.endswith(".md"):
        file_path = os.path.join(KNOWLEDGE_BASE_DIR, filename)

        # 读取文件内容
        with open(file_path, "r", encoding="utf-8") as f:
            content = f.read()

        # 生成嵌入向量
        # 注意：如果文件内容过长，需要分块处理
        embedding = llm.create_embedding(content)

        # 将向量和原始文本存入数据库
        vector_db.append({
            "filename": filename,
            "content": content,
            "embedding": embedding["embedding"]
        })

        print(f"已处理: {filename}")

# ==========================================
# 4. 保存向量数据库
# ==========================================
print(f"正在保存向量数据库到: {OUTPUT_VECTOR_DB} ...")
with open(OUTPUT_VECTOR_DB, "w", encoding="utf-8") as f:
    json.dump(vector_db, f, ensure_ascii=False, indent=2)

print("✅ 嵌入处理完成！")
